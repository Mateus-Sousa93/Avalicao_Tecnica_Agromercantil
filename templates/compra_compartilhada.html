Schema - Compra Compartilhada

1. Tabela: grupos_compra

CREATE TABLE grupos_compra (
    id_grupo SERIAL PRIMARY KEY,
    nome_grupo VARCHAR(200) NOT NULL,
    tipo_grupo VARCHAR(50) CHECK (tipo_grupo IN 
        ('COOPERATIVA', 'ASSOCIACAO', 'CONSORCIO', 'GRUPO_INFORMAL')),
    representante_id INTEGER REFERENCES clientes(id_cliente),
    data_criacao DATE DEFAULT CURRENT_DATE,
    ativo BOOLEAN DEFAULT TRUE
);

2. Tabela: cliente_grupo

CREATE TABLE cliente_grupo (
    id_cliente INTEGER REFERENCES clientes(id_cliente),
    id_grupo INTEGER REFERENCES grupos_compra(id_grupo),
    percentual_participacao NUMERIC(5,2) 
        CHECK (percentual_participacao > 0 AND percentual_participacao <= 100),
    data_entrada DATE DEFAULT CURRENT_DATE,
    ativo BOOLEAN DEFAULT TRUE,
    PRIMARY KEY (id_cliente, id_grupo)
);

3. Tabela: pedido_coproprietario

CREATE TABLE pedido_coproprietario (
    id_pedido INTEGER REFERENCES pedidos(id_pedido),
    id_cliente INTEGER REFERENCES clientes(id_cliente),
    percentual_propriedade NUMERIC(5,2) NOT NULL 
        CHECK (percentual_propriedade > 0 AND percentual_propriedade <= 100),
    valor_proporcional NUMERIC(15,2) NOT NULL,
    tipo_participacao VARCHAR(50) CHECK (tipo_participacao IN 
        ('COMPRADOR', 'FINANCIADOR', 'REPRESENTANTE')),
    PRIMARY KEY (id_pedido, id_cliente)
);

4. Função: criar_pedido_compartilhado()

CREATE OR REPLACE FUNCTION criar_pedido_compartilhado(
    p_data_pedido DATE,
    p_data_entrega DATE,
    p_tipo_contrato VARCHAR(50),
    p_regiao_origem VARCHAR(100),
    p_regiao_destino VARCHAR(100),
    p_proprietarios JSONB
) RETURNS INTEGER AS $$
DECLARE
    v_id_pedido INTEGER;
    v_proprietario RECORD;
BEGIN
    -- Validar que percentuais somam 100
    IF (SELECT COALESCE(SUM((value->>'percentual')::NUMERIC), 0) 
        FROM jsonb_array_elements(p_proprietarios)) != 100 THEN
        RAISE EXCEPTION 'Percentuais devem somar 100%%';
    END IF;
    
    -- Criar pedido
    INSERT INTO pedidos (...) VALUES (...)
    RETURNING id_pedido INTO v_id_pedido;
    
    -- Inserir co-proprietários
    FOR v_proprietario IN ... LOOP
        INSERT INTO pedido_coproprietario (...)
        VALUES (...);
    END LOOP;
    
    RETURN v_id_pedido;
END;
$$ LANGUAGE plpgsql;
